// Példa frissítő lekérdezés
string updateQuery = "UPDATE table_name SET column1 = @value1, column2 = @value2 WHERE id = @id;";

var parameters = new Dictionary<string, object>
{
    { "@value1", "Új első érték" },
    { "@value2", "Új második érték" },
    { "@id", 1 } // A frissítendő sor azonosítója
};

Update(updateQuery, parameters);
